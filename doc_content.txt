export default async function handler(req, res) {
  const dados = req.body; // Recebe os dados
  
  // 1. Gera o PDF em memória (buffer), NUNCA salva em disco/banco
  const pdfBuffer = await gerarPDF(dados);

  res.setHeader('Content-Type', 'application/pdf');

  // Nenhum log, nenhum banco, nenhum cache

// Criptografa dados sensíveis antes de salvar
function criptografar(texto, chave) {
  const iv = crypto.randomBytes(16);
  const cipher = crypto.createCipheriv('aes-256-gcm', chave, iv);
  return iv + cipher.update(texto, 'utf8', 'hex') + cipher.final('hex');
}

1. Auto-save React Hook (ou Vanilla JS)
// hooks/useFormAutoSave.js
import { useEffect, useState } from 'react';

export function useFormAutoSave(formKey) {
  const [data, setData] = useState(() => {
    // Recupera do localStorage ao carregar
    const saved = localStorage.getItem(`sindicato_draft_${formKey}`);
    return saved ? JSON.parse(saved) : {};
  });

  useEffect(() => {
    // Salva a cada alteração (debounce de 500ms ideal)
    const timer = setTimeout(() => {
      localStorage.setItem(`sindicato_draft_${formKey}`, JSON.stringify(data));
    }, 500);
    
    return () => clearTimeout(timer);
  }, [data, formKey]);

  const clearDraft = () => {
    localStorage.removeItem(`sindicato_draft_${formKey}`);
  };

  return { data, setData, clearDraft };
}

// Uso no componente:
function FormCadastro() {
  const { data, setData, clearDraft } = useFormAutoSave('cadastro_lazinho');
  
  const handleChange = (e) => {
    setData(prev => ({ ...prev, [e.target.name]: e.target.value }));
  };
  
  const gerarPDF = async () => {
    // Envia dados para API gerar PDF
    await fetch('/api/gerar-pdf', {
      method: 'POST',
      body: JSON.stringify(data)
    });
    
    // LIMPA TUDO após download
    clearDraft();
  };
  
  return (
    <input 
      name="telefone" 
      value={data.telefone || ''} 
      onChange={handleChange}
      placeholder="Telefone"
    />
  );
}

Os dados do formulário (Nome, Pai, Mãe, Endereço, RG, Telefone) nunca entram no banco. Eles vão direto para o PDF.

const cpfHashHex = Array.from(new Uint8Array(cpfHash)).map(b => b.toString(16).padStart(2, '0')).join('');

Resposta direta à sua dúvida
“Como o usuário está vinculado ao PDF se não salvamos os dados?”
Você salva apenas uma “impressão digital” (hash) do CPF ou um ID aleatório (UUID) gerado no momento do cadastro. O PDF carrega esse mesmo identificador (invisível para o usuário, como metadado ou watermark), permitindo que o sistema saiba “este PDF pertence a este usuário” sem saber o conteúdo do PDF.

A Solução: “Cofre Digital”
Você não salva os dados (nome, CPF, endereço), mas pode salvar o arquivo PDF criptografado como um “blob” opaco (um arquivo morto que só o usuário pode abrir).
Como funciona:
// 1. Gera o PDF (com todos os dados sensíveis)
const pdfBytes = await gerarPDF(dadosCompletos);

// 2. Deriva uma chave da senha do usuário (ele precisa criar uma senha no cadastro)
const chave = await pbkdf2(senhaDoUsuario, salt, 100000, 32, 'sha256');

// 3. Criptografa o PDF inteiro (AES-256)
const pdfCriptografado = await aesEncrypt(pdfBytes, chave);

// 4. Salva NO BANCO apenas isso:
// tabela: documentos
// - user_id: uuid
// - file_blob: pdfCriptografado (ilegível sem a senha)
// - created_at: timestamp
// - filename_hash: hash(nome_arquivo) (para identificar sem sabber o conteúdo)
O que está no seu banco: - Um arquivo binário criptografado (parece lixo aleatório) - Sem a senha do usuário, é impossível abrir - Você não sabe o que tem dentro (nem nome, nem CPF, nada)
Quando o usuário loga: 1. Ele digita a senha 2. Sistema pega o blob criptografado do banco 3. Descriptografa no navegador (client-side) usando a senha dele 4. Ele baixa o PDF legível novamente
Por que isso é seguro?
Sem a senha do usuário (que nunca sai do navegador durante a descriptografia), o PDF é inútil.
Alternativa: “Download Único + Responsabilidade do Usuário”
Se quiser manter a promessa de “não guardamos nada, absolutamente nada”:
⚠️ AVISO NA TELA:
"Baixe o PDF agora. Não salvamos cópias por segurança. 
Se perder o arquivo, precisará fazer um novo cadastro na secretaria."
Vantagens: - Zero risco de vazamento (não existe o que vazar) - Custo zero de storage - Compliance total com LGPD (você literalmente não tem os dados)
Desvantagens: - Se o usuário perde o celular/computador, perde o comprovante - Mas isso é um recurso de segurança, não um bug

Dado que é um documento oficial de cadastro sindical, eu sugiro a Solução do Cofre Digital (PDF criptografado), porque:

Você não precisa dos dados em texto, só do arquivo “fechado”
A chave está com o usuário (zero responsabilidade legal para o sindicato em caso de vazamento)

Características: - Irreversível: Não dá para descobrir “Maria123!” olhando o hash - Único: Cada usuário tem um hash diferente mesmo com a mesma senha (salts aleatórios) - Lento: bcrypt consome tempo de propósito (dificulta brute force)
O que NUNCA acontece:
❌ Salvar senha em texto puro
❌ Salvar senha “criptografada” (criptografia é reversível, hash não)
❌ Enviar senha por email
4. Estrutura do Banco (Mínima)
-- Tabela 1: Autenticação (não contém dados pessoais)
CREATE TABLE users (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    email VARCHAR(255) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL, -- bcrypt
    cpf_hash VARCHAR(64) UNIQUE, -- SHA-256 para busca sem expor CPF
    created_at TIMESTAMP DEFAULT NOW()
);

-- Tabela 2: Documentos (blobs criptografados)
CREATE TABLE documents (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES users(id),
    encrypted_blob BYTEA NOT NULL, -- PDF criptografado (ilegível)
    iv VARCHAR(32) NOT NULL, -- Vetor de inicialização AES
    salt VARCHAR(32) NOT NULL, -- Salt do PBKDF2
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP
);

-- Índice para performance (não expõe dados)
CREATE INDEX idx_docs_user ON documents(user_id);
O que NÃO existe neste banco: - Coluna nome, endereco, telefone, rg - CPF em texto - PDF em texto (apenas criptografado)
5. Fluxo Completo: Cadastro → Download
ETAPA 1: Cadastro (Secretaria)
// 1. Coleta dados (formulário)
const dados = {
  nome: "Claudia Maria",
  cpf: "648.666.456-87",
  // ... todos os outros campos
  senha: "Segura@2025" // Criada no momento do cadastro
};

// 2. Gera PDF em memória (Serverless)
const pdfBuffer = await gerarPDF(dados);

// 3. Deriva chave criptográfica da senha
const salt = crypto.randomBytes(16);
const chaveAES = await pbkdf2(dados.senha, salt, 100000, 32, 'sha256');

// 4. Criptografa o PDF
const iv = crypto.randomBytes(16);
const cipher = crypto.createCipheriv('aes-256-gcm', chaveAES, iv);
const encryptedPDF = Buffer.concat([cipher.update(pdfBuffer), cipher.final()]);
const authTag = cipher.getAuthTag(); // Garante integridade

// 5. Prepara dados para banco
const userData = {
  email: dados.email,
  password_hash: await bcrypt.hash(dados.senha, 12), // Para login
  cpf_hash: sha256(dados.cpf) // Para busca futura
};

const docData = {
  encrypted_blob: encryptedPDF,
  iv: iv.toString('hex'),
  salt: salt.toString('hex'),
  auth_tag: authTag.toString('hex')
};

// 6. Salva no PostgreSQL (dados sensíveis são descartados da memória)
await db.query('INSERT INTO users...', userData);
await db.query('INSERT INTO documents...', docData);

// 7. LIMPA tudo (GC remove dados da RAM)
dados = null; 
pdfBuffer = null;
chaveAES = null;
ETAPA 2: Login + Download (Associado)
// 1. Usuário digita email/senha
const { email, senha } = req.body;

// 2. Busca usuário
const user = await db.query('SELECT * FROM users WHERE email = $1', [email]);

// 3. Valida senha (compara hash)
const valido = await bcrypt.compare(senha, user.password_hash);
if (!valido) return erro(401);

// 4. Busca documento criptografado
const doc = await db.query('SELECT * FROM documents WHERE user_id = $1', [user.id]);

// 5. DERIVA a mesma chave usando a senha digitada + salt salvo
const salt = Buffer.from(doc.salt, 'hex');
const chaveAES = await pbkdf2(senha, salt, 100000, 32, 'sha256');

// 6. DECRIPTOGRAFA no servidor (ou envia para client-side)
const decipher = crypto.createDecipheriv('aes-256-gcm', chaveAES, Buffer.from(doc.iv, 'hex'));
decipher.setAuthTag(Buffer.from(doc.auth_tag, 'hex'));
const pdfOriginal = Buffer.concat([decipher.update(doc.encrypted_blob), decipher.final()]);

// 7. Envia para download
res.setHeader('Content-Type', 'application/pdf');
res.send(pdfOriginal);

// 8. LIMPA memória
pdfOriginal.fill(0); // Sobrescreve com zeros (secure wipe)
